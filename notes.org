quantize :: Int -> Int -> Int
quantize acc n | n < (acc * res) = acc * res
quantize acc n | n > (acc * res) && n < (acc * res + (res / 2)) = acc * res
quantize acc n | n > (acc * res) && n > (acc * res + (res / 2)) = quantize (acc + 1) n
